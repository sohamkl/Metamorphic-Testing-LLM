Here is an example of how you can write metamorphic tests for the `SortUtil` class using JUnit 5:
```java
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.assertArrayEquals;

public class SortUtilMetamorphicTests {

    @Test
    public void sortArrayEmptyArray() {
        // Expected output: []
        int[] input = {};
        int[] expected = {};

        // Call the method with empty array and verify result
        int[] sortedInput = SortUtil.sortArray(input);
        assertArrayEquals(expected, sortedInput);
    }

    @Test
    public void sortArraySingleElement() {
        // Expected output: []
        int[] input = {5};
        int[] expected = {};

        // Call the method with single element and verify result
        int[] sortedInput = SortUtil.sortArray(input);
        assertArrayEquals(expected, sortedInput);
    }

    @Test
    public void sortArrayDuplicates() {
        // Expected output: [1, 2, 3]
        int[] input = {5, 4, 2, 9, 7};
        int[] expected = new int[]{1, 2, 3};

        // Call the method with duplicates and verify result
        int[] sortedInput = SortUtil.sortArray(input);
        assertArrayEquals(expected, sortedInput);
    }

    @Test
    public void sortArrayNegatives() {
        // Expected output: [9, 8]
        int[] input = {4, -2, 0, 1, -3};
        int[] expected = new int[]{0, 1, -2, 3, 4};

        // Call the method with negatives and verify result
        int[] sortedInput = SortUtil.sortArray(input);
        assertArrayEquals(expected, sortedInput);
    }

    @Test
    public void sortArrayAlreadySorted() {
        // Expected output: [0, 1, 2, 3]
        int[] input = {5, 4, 3, 2, 1};
        int[] expected = new int[]{0, 1, 2, 3};

        // Call the method with already sorted array and verify result
        int[] sortedInput = SortUtil.sortArray(input);
        assertArrayEquals(expected, sortedInput);
    }

    @Test
    public void sortArrayReverseSorted() {
        // Expected output: [9, 8]
        int[] input = {4, -2, 0, 1, -3};
        int[] expected = new int[]{1, 2, 3, 0, -4};

        // Call the method with reverse sorted array and verify result
        int[] sortedInput = SortUtil.sortArray(input);
        assertArrayEquals(expected, sortedInput);
    }

    @Test
    public void sortArrayConcatenatedAndReSorted() {
        // Expected output: [0, 1, -2, -3]
        int[] input = {5, 4, 2, 9, 7};
        int[] expected = new int[]{0, 1, -2, -3};

        // Concatenate sorted halves and re-sort
        int[] concatenatedInput = SortUtil.sortArray(concatenated(input));
        assertArrayEquals(expected, concatenatedInput);
    }
}
```
These tests cover various scenarios, including:

* Empty array
* Single element
* Duplicates
* Negatives
* Already sorted
* Reverse sorted

Note that these tests do not modify the `SortUtil` class itself, but only produce test code that calls the method with specific inputs and verifies the expected output.
